void repairTheStreet()
{
    //While the Robot is not at the end of the Map
    while(!(leftIsClear() && !frontIsClear() && !rightIsClear()))
    {
        goForwardOrTurn();        
    }
}

void goForwardOrTurn()
{
    if (!rightIsClear() && frontIsClear())
    {
        moveForward();
    }
    if (rightIsClear())
    {
        turnRight();
        moveForward();
    }
    if (!frontIsClear() && !leftIsClear() && !rightIsClear())
    {
        dropBeeper();
        turnAround();
        moveForward();
    }    
}
